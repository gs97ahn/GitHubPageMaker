Population Data using Vertical Bar Graph

import csv
import matplotlib.pyplot as plt

f = open('incheon_population.csv', 'r', encoding='cp949')
data = csv.reader(f)

result = []

for row in data:
    if '송도1동' in row[0]:
        for m, fm in zip(row[3:104], row[106:207]):
            result.append(int(m) + int(fm))
            
f.close()

plt.rc('font', family='AppleGothic')
plt.title('송도1동 연령별 인구수')

plt.bar(range(101), result)
plt.xlabel('연령')
plt.ylabel('인구수')
plt.show()

Population by Gender using Horizontal Bar Graph

import csv
import matplotlib.pyplot as plt

f = open('incheon_population.csv', 'r', encoding='cp949')
data = csv.reader(f)

male = []
female = []

for row in data:
    if '송도1동' in row[0]:
        for i in row[3:104]:
            male.append(-int(i))
        for i in row[106:207]:
            female.append(int(i))

f.close()

plt.rc('font', family='AppleGothic')
plt.title('송도1동 연령별 인구수')

plt.rcParams['axes.unicode_minus'] = False
plt.barh(range(101), male, label='남성')
plt.barh(range(101), female, label='여성')
plt.xlabel('연령')
plt.ylabel('인구수')
plt.legend()
plt.show()

Population by Gender using Pie

import csv
import matplotlib.pyplot as plt

f = open('incheon_population.csv', 'r', encoding='cp949')
data = csv.reader(f)

result = []
m_sum, f_sum = 0, 0

for row in data:
    if '송도1동' in row[0]:
        for i in range(101):
            m_sum += int(row[i+3])
            f_sum += int(row[i+106])
            
result.append(m_sum)
result.append(f_sum)

plt.rc('font', family='AppleGothic')
plt.axis('equal')

label = ['남', '여']
color = ['crimson', 'darkcyan']
plt.pie(result, labels=label, autopct='%.1f%%', colors=color, startangle=90)
plt.legend()
plt.show()

Population by Gender and Age Group using Pie

import csv
import matplotlib.pyplot as plt
import math

f = open('incheon_population.csv', 'r', encoding='cp949')
data = csv.reader(f)

male, female = [], []
size = []

for row in data:
    if '송도1동' in row[0]:
        for i in range(3, 104):
            m, fm = int(row[i]), int(row[i+103])
            male.append(m)
            female.append(fm)
            size.append(math.sqrt(m+fm))

f.close()

plt.rc('font', family='AppleGothic')
plt.figure(figsize=(10, 5), dpi=300)
plt.title('연령대별 성별 인구 그래프', fontsize=15, color='b')
plt.rcParams['axes.unicode_minus'] = False

plt.scatter(male, female, c=range(101), alpha=0.9, cmap='jet')
plt.colorbar()

max_val = max(max(male), max(female))
plt.axis([0, max_val, 0, max_val])
line = range(max_val)
plt.plot(line, line, color='g')
plt.xlabel('남성 인구수')
plt.ylabel('여성 인구수')
plt.show()
